# Temperature (LM35)

Run with:
```bash
node eg/temperature-lm35.js
```
<!--remove-end-->

```javascript
var five = require("johnny-five");

five.Board().on("ready", function() {
  var temperature = new five.Temperature({
    controller: "LM35",
    pin: "A0"
  });

  temperature.on("data", function(data) {
    console.log(data.celsius + "°C", data.fahrenheit + "°F");
  });
});


```
